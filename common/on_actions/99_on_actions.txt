on_game_started = {
	effect = {
		# Memento Mori
		every_country = {
			every_interest_group = {
				leader = {
					if = {
						limit = {
							NOT = {
								has_variable = memento_mori_immunity
							}
						}
						set_variable = {
							name = memento_mori_immunity
							years = 5
						}
					}
				}
			}
		}
	}
	# above is necessary from vanilla, on_game_started always overwrites
	on_actions = {
		calculate_literacy_rate_on_action
	}
}
